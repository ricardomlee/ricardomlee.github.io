#### 1. 玩游戏 （01背包）

> 输入第一项：case数；第二行：游戏个数N，总时间X；后面N行：游戏价值，游戏时间
输出：最大总价值
输入
1
3 5
20 2
10 3
30 2
输出
50


```C++
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int gamegain(int T, vector<int> time, vector<int> gain)
{
    vector<int> maxgain(T+1);
    for(int i=0;i<time.size();++i){
        for(int j=T;j>=0;--j){
            if(j>=time[i]){
                maxgain[j]=max(gain[i]+maxgain[j-time[i]],maxgain[j]);
            }
            else
            {
                maxgain[j]=maxgain[j]?maxgain[j]:0;
            }
        }
    }
    return maxgain[T];
}

int main()
{
    vector<int> final_res;
    int T;
    cin>>T;
    for(int i=0;i<T;++i){
        int N,X;
        cin>>N>>X;
        vector<int> A;
        vector<int> B;
        for(int j=0;j<N;++j){
            int a,b;
            cin>>a>>b;
            A.push_back(a);
            B.push_back(b);
        }

        final_res.push_back(gamegain(X,B,A));
    }

    for(int i=0;i<final_res.size();++i){
        cout<<final_res[i]<<endl;
    }
    return 0;
}
```

#### 2.机器人 (map)
> 输入 多组测试数据，第一行：映射命令数目N，测试命令数目M；
cmd action
cmd action
cmd action
cmd_test
cmd_test
输出
action_test
action_test

```C++
#include <iostream>
#include <vector>
#include <string>
#include <map>

using namespace std;

int main()
{
    vector<string> res;
	int N, M;
	while (cin >> N >> M) {
		map<string, string> cmdmap;
		for (int i = 0; i<N; ++i) {
			string cmd, action;
			cin >> cmd >> action;
			cmdmap.insert(pair<string, string>(cmd, action));
		}
		for (int i = 0; i<M; ++i) {
			string newcmd;
			cin >> newcmd;
			auto found = cmdmap.find(newcmd);
			res.push_back(found->second);
		}
	}

	for(int i=0;i<res.size();++i){
        cout<<res[i]<<endl;
    }
	return 0;
}
```
